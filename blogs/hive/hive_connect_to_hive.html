    /**
     * å°†ResultSetç»“æœé›†éå†åˆ°Listä¸­
     *
     * @param rs
     * @return
     * @throws SQLException
     */
    private static List<Map<String, Object>> convertList(ResultSet rs) throws SQLException {
        List<Map<String, Object>> list = new ArrayList<>();
        ResultSetMetaData md = rs.getMetaData();// è·å–é”®å
        int columnCount = md.getColumnCount();// è·å–è¡Œçš„æ•°é‡
        while (rs.next()) {
            Map<String, Object> rowData = new HashMap<>();// å£°æ˜Map
            for (int i = 1; i <= columnCount; i++) {
                rowData.put(md.getColumnName(i), rs.getObject(i));// è·å–é”®ååŠå€¼
            }
            list.add(rowData);
        }
        return list;
    }
