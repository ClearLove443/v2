        response.setHeader("Access-Control-Allow-Methods", "POST, GET, PATCH, DELETE, PUT");

        response.setHeader("Access-Control-Allow-Headers", "Origin, X-Requested-With, Content-Type, Accept");
        chain.doFilter(req, res);
    }
}
